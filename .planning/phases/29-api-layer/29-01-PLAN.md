**api/openapi.yaml — Full OpenAPI 3.0 specification:**

```yaml
openapi: 3.0.3
info:
  title: ShiftWell Enterprise API
  version: 1.0.0
  description: |
    REST API for hospital scheduling system integrations.
    Push shift schedules, pull anonymized outcome data.
    All data de-identified per HIPAA Safe Harbor (45 CFR §164.514(b)).
  contact:
    name: ShiftWell Enterprise Support
    email: enterprise@shiftwell.app

servers:
  - url: https://api.shiftwell.app/v1
    description: Production
  - url: http://localhost:3002/v1
    description: Local development

security:
  - BearerAuth: []

components:
  securitySchemes:
    BearerAuth:
      type: http
      scheme: bearer
      bearerFormat: JWT
    OAuth2:
      type: oauth2
      flows:
        clientCredentials:
          tokenUrl: /oauth/token
          scopes:
            schedule:write: Push shift schedules
            outcomes:read: Pull outcome metrics

  schemas:
    ShiftRecord:
      type: object
      required: [employeeId, shiftType, startTime, endTime]
      properties:
        employeeId: { type: string, description: "Opaque employee identifier (not name/SSN)" }
        shiftType: { type: string, enum: [day, evening, night, rotating] }
        startTime: { type: string, format: date-time }
        endTime: { type: string, format: date-time }
        unit: { type: string, description: "Optional: ICU, ED, Med-Surg, etc." }

    SchedulePushRequest:
      type: object
      required: [orgId, schedules, format]
      properties:
        orgId: { type: string }
        format: { type: string, enum: [kronos, qgenda, shiftwell] }
        schedules: { type: array, items: { $ref: "#/components/schemas/ShiftRecord" } }
        periodStart: { type: string, format: date }
        periodEnd: { type: string, format: date }

    CohortMetrics:
      type: object
      properties:
        orgId: { type: string }
        cohortSize: { type: integer }
        avgRecoveryScore: { type: number, minimum: 0, maximum: 100 }
        adherenceRate: { type: number, minimum: 0, maximum: 1 }
        avgDebtBalance: { type: number, description: "Minutes of sleep debt" }
        p25RecoveryScore: { type: number }
        p75RecoveryScore: { type: number }
        lowRecoveryWorkerPct: { type: number }
        periodStart: { type: string, format: date }
        periodEnd: { type: string, format: date }

    OutcomesResponse:
      type: object
      properties:
        metadata:
          type: object
          properties:
            orgId: { type: string }
            generatedAt: { type: string, format: date-time }
            cohortSize: { type: integer }
            dpApplied: { type: boolean }
        metrics: { $ref: "#/components/schemas/CohortMetrics" }

    ErrorResponse:
      type: object
      properties:
        error: { type: string }
        code: { type: string }
        requestId: { type: string }

paths:
  /oauth/token:
    post:
      summary: Get access token (client credentials)
      security: []
      requestBody:
        content:
          application/x-www-form-urlencoded:
            schema:
              properties:
                grant_type: { type: string, enum: [client_credentials] }
                client_id: { type: string }
                client_secret: { type: string }
                scope: { type: string }
      responses:
        "200":
          content:
            application/json:
              schema:
                properties:
                  access_token: { type: string }
                  token_type: { type: string, example: bearer }
                  expires_in: { type: integer, example: 3600 }
        "401":
          content: { application/json: { schema: { $ref: "#/components/schemas/ErrorResponse" } } }

  /schedules:
    post:
      summary: Push shift schedules
      security: [{ BearerAuth: [] }]
      requestBody:
        required: true
        content:
          application/json:
            schema: { $ref: "#/components/schemas/SchedulePushRequest" }
      responses:
        "201":
          description: Schedules accepted
          content:
            application/json:
              schema:
                properties:
                  accepted: { type: integer }
                  rejected: { type: integer }
                  requestId: { type: string }
        "400": { content: { application/json: { schema: { $ref: "#/components/schemas/ErrorResponse" } } } }
        "401": { content: { application/json: { schema: { $ref: "#/components/schemas/ErrorResponse" } } } }
        "429": { description: Rate limit exceeded }

  /outcomes:
    get:
      summary: Pull cohort outcome metrics
      security: [{ BearerAuth: [] }]
      parameters:
        - name: start
          in: query
          required: true
          schema: { type: string, format: date }
        - name: end
          in: query
          required: true
          schema: { type: string, format: date }
        - name: period
          in: query
          schema: { type: string, enum: [daily, weekly, monthly], default: monthly }
        - name: format
          in: query
          schema: { type: string, enum: [json, csv], default: json }
      responses:
        "200":
          content:
            application/json:
              schema: { $ref: "#/components/schemas/OutcomesResponse" }
            text/csv:
              schema: { type: string }
        "401": { content: { application/json: { schema: { $ref: "#/components/schemas/ErrorResponse" } } } }
        "429": { description: Rate limit exceeded }
```

**api/src/types/api.ts:**
TypeScript types mirroring the OpenAPI schemas. KronosShiftRecord and QGendaShiftRecord formats for parser.

**api/src/middleware/auth.ts:**
Express middleware. Verifies Bearer JWT. Decodes orgId from token claims. Adds `req.orgId` to request. Uses `jose` library (not jsonwebtoken — Edge runtime compatible). JWT signed with RS256. Rejects tokens where `orgId` claim is missing or doesn't match request's orgId parameter (scope enforcement).

**api/src/middleware/rateLimit.ts:**
In-memory rate limiter. 100 requests per minute per client_id (from JWT sub claim). Returns 429 with `Retry-After` header when exceeded. Use sliding window algorithm (not fixed window).

**api/src/routes/schedule.ts:**

POST /v1/schedules handler:
1. Extract orgId from req.orgId (set by auth middleware)
2. Validate request body against SchedulePushRequest schema (zod)
3. Parse based on format:
   - 'shiftwell': direct ShiftRecord[] — no transformation needed
   - 'kronos': KronosParser.parse(schedules) — handles Kronos export format. Kronos uses: `EmployeeNumber` (→ employeeId), `AssignedShiftCode` (day/evening/night/rotating mapping), `ScheduleDate` + `StartTime`/`StopTime` (→ startTime/endTime ISO)
   - 'qgenda': QGendaParser.parse(schedules) — QGenda uses: `EmployeeId`, `TaskName` (maps to shiftType via lookup), `StartDate`/`EndDate`
4. Write parsed records to DB (use SQLite/better-sqlite3 for local dev, PostgreSQL for prod — abstract behind ScheduleRepository interface)
5. Return { accepted: N, rejected: M, requestId: uuid() }

**api/src/routes/outcomes.ts:**

GET /v1/outcomes handler:
1. Validate query params: start (required), end (required), period, format
2. Extract orgId from req.orgId
3. Fetch UserRecord[] for orgId from DB (UserRepository.findByOrgId(orgId, { start, end }))
4. Call buildCohortMetrics(users, { epsilon: 1.0, sensitivity: 100, cohortThreshold: 50 }, { start, end })
5. If format=csv: set Content-Type: text/csv, return toCSV(result)
6. If format=json (default): return toJSON(result)
