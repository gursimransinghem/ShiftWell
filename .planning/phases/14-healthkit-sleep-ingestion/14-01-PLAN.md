    **Step 1: Add getSleepHistoryForRange to healthkit-service.ts**

    After the existing `getSleepHistory` function, add:

    ```typescript
    /**
     * Read sleep records for the last N nights (for discrepancy history).
     * Returns up to 30 records. Uses asleepStart (not inBedStart) per Phase 13
     * research: asleepStart is more accurate for feedback timing (avoids
     * pre-sleep reading-in-bed latency skewing the deviation calculation).
     *
     * @param nights - Number of nights to look back (default: 30, max: 30)
     */
    export async function getSleepHistoryForRange(nights: number = 30): Promise<SleepRecord[]> {
      const available = await isAvailable();
      if (!available) return [];

      const endDate = new Date();
      const startDate = subDays(endDate, Math.min(nights, 30));

      try {
        const records = await getSleepHistory(startDate, endDate);
        // Filter to records with asleepStart (required for feedback timing)
        return records.filter(r => r.asleepStart !== null);
      } catch {
        return [];
      }
    }
    ```

    // Discrepancy history for feedback engine (HK-01, HK-02, HK-03)
    const thirtyNightHistory = await getSleepHistoryForRange(30);
    const currentPlan = usePlanStore.getState().plan;

    if (currentPlan && thirtyNightHistory.length > 0) {
      const comparisons = thirtyNightHistory.map(record => {
        // Find the planned sleep window for this night's date
        const nightDate = record.asleepStart ?? record.date;
        const plannedDay = currentPlan.days?.find(
          d => format(new Date(d.date), 'yyyy-MM-dd') === format(nightDate, 'yyyy-MM-dd')
        );
        const mainSleepBlock = plannedDay?.blocks?.find(b => b.type === 'sleep');

        if (!mainSleepBlock) {

          return comparePlannedVsActual(nightDate, nightDate, null);

        return comparePlannedVsActual(
          new Date(mainSleepBlock.start),
          new Date(mainSleepBlock.end),
          record

      setDiscrepancyHistory(comparisons);
    } else if (thirtyNightHistory.length === 0) {
      // HK unavailable or no data — set empty, feedback engine will pause
      setDiscrepancyHistory([]);
    }
