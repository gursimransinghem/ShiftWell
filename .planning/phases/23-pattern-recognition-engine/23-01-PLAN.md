    export type PatternType =
      | 'consecutive-night-impact'
      | 'debt-trend-rising'

      | 'weekend-compensation'
      | 'chronic-late-bedtime'
      | 'shift-transition-cluster';

    export type PatternSeverity = 'info' | 'warning' | 'critical';

    export interface DetectedPattern {
      id: string;                     // UUID
      type: PatternType;
      severity: PatternSeverity;
      detectedAtISO: string;
      windowStartISO: string;         // Start of analysis window
      windowEndISO: string;
      metadata: Record<string, number | string>;  // Pattern-specific data
    }

    export interface PatternAlert {

    **Step 2: Create `src/lib/patterns/pattern-detector.ts`**

    Export `detectPatterns(history: DiscrepancyHistory, shifts: ShiftRecord[]): DetectedPattern[]`

    Implement three detectors:

    **Detector 1: Consecutive Night Shift Impact (PAT-01)**
    - Scan shifts array for runs of consecutive nights (type==='night')
    - Trigger at 3+ consecutive nights
    - metadata: `{ nightCount: N, avgSleepQualityDrop: X }` (quality drop = mean discrepancy delta for those nights vs baseline)
    - severity: 3-4 nights = warning, 5+ = critical

    **Detector 2: Recovery Debt Trend (PAT-02)**
    - Divide last 28 days into 4 weekly buckets
    - Calculate mean absolute discrepancy per week
    - If week 4 > week 1 by > 0.5h: direction = 'rising' (warning)
    - If week 4 < week 1 by > 0.5h: direction = 'improving' (info)
    - metadata: `{ week1AvgHours: X, week4AvgHours: X, direction: 'rising'|'improving'|'stable' }`

    **Detector 3: Weekend Compensation**
    - Group records by day-of-week
    - Compute mean sleep duration for Mon-Fri vs Sat-Sun
    - If weekend mean > weekday mean by 90+ min: trigger
    - metadata: `{ weekdayAvgHours: X, weekendAvgHours: X, excessMinutes: N }`
    - severity: 90-120 min excess = warning, 120+ = critical (social jetlag signal)

    Return deduplicated array sorted by severity (critical first).

    Write all 8 failing tests first (TDD red phase).
