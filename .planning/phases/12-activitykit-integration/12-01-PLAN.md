    ```swift
    import Foundation

    struct ShiftWellLiveActivityAttributes: ActivityAttributes {
        public typealias ShiftWellState = ContentState

        /// Static data — does not change during the activity
        public struct ContentState: Codable, Hashable {
            /// Current phase of the sleep cycle
            var phase: String           // "wind-down" | "sleep" | "morning"
            /// Countdown in minutes (wind-down and morning phases)
            var countdownMinutes: Int?
            /// Recovery score 0-100 (morning phase only)
            var score: Int?
            /// Short label for compact island view
            var label: String
        }

        /// Static: the user's name shown in expanded view
        var userName: String
    }
    ```

    **Step 3: Create ShiftWellLiveActivityView.swift**

    Dynamic Island compact + expanded views. Uses SwiftUI only — no UIKit.

    ```swift
    import ActivityKit
    import SwiftUI
    import WidgetKit

    struct ShiftWellLiveActivityView: Widget {
        var body: some WidgetConfiguration {
            ActivityConfiguration(for: ShiftWellLiveActivityAttributes.self) { context in
                // Lock screen / StandBy banner
                HStack {
                    Image(systemName: phaseIcon(context.state.phase))
                        .foregroundColor(.purple)
                    Text(context.state.label)
                        .font(.caption)
                        .foregroundColor(.white)
                }
                .padding(.horizontal, 16)
                .padding(.vertical, 8)
                .background(.black.opacity(0.8))
                .cornerRadius(12)

            } dynamicIsland: { context in
                DynamicIsland {
                    // Expanded view
                    DynamicIslandExpandedRegion(.leading) {
                        Image(systemName: phaseIcon(context.state.phase))
                            .foregroundColor(.purple)
                    }
                    DynamicIslandExpandedRegion(.center) {
                        Text(context.state.label)
                            .font(.caption2)
                            .foregroundColor(.white)
                    }
                    DynamicIslandExpandedRegion(.trailing) {
                        if let score = context.state.score {
                            Text("\(score)")
                                .font(.caption.bold())
                                .foregroundColor(scoreColor(score))
                        } else if let mins = context.state.countdownMinutes {
                            Text("\(mins)m")
                                .font(.caption2)
                                .foregroundColor(.gray)
                        }
                    }
                } compactLeading: {
                    Image(systemName: phaseIcon(context.state.phase))
                        .foregroundColor(.purple)
                        .font(.caption2)
                } compactTrailing: {
                    if let mins = context.state.countdownMinutes, context.state.phase != "morning" {
                        Text("\(mins)m")
                            .font(.caption2)
                            .foregroundColor(.white)
                    } else if let score = context.state.score {
                        Text("\(score)")
                            .font(.caption2.bold())
                            .foregroundColor(scoreColor(score))
                    }
                } minimal: {
                    Image(systemName: phaseIcon(context.state.phase))
                        .foregroundColor(.purple)
                        .font(.caption2)
                }
            }
        }

        private func phaseIcon(_ phase: String) -> String {
            switch phase {
            case "wind-down": return "moon.zzz"
            case "sleep":     return "moon.fill"
            case "morning":   return "sun.and.horizon"
            default:          return "moon"
            }
        }

        private func scoreColor(_ score: Int) -> Color {
            score >= 70 ? .green : score >= 50 ? .yellow : .red
        }
    }
    ```

    ```typescript
    export async function updateSleepActivity(state: LiveActivityState): Promise<void> {
      if (LIVE_ACTIVITIES_AVAILABLE) {
        await LiveActivity.update({
          activityType: 'ShiftWellLiveActivity',
          contentState: {
            phase: state.phase,
            countdownMinutes: state.countdownMinutes ?? null,
            score: state.score ?? null,
            label: state.label,
          },
        });
        return;
      }
      // Fallback: update notification (reuse start logic with new label)
      await startSleepActivity(state);
    }

    export async function endSleepActivity(): Promise<void> {
      if (LIVE_ACTIVITIES_AVAILABLE) {
        await LiveActivity.end({ activityType: 'ShiftWellLiveActivity' });
        return;
      }
      // Fallback: cancel all live-activity notifications
      await Notifications.cancelAllScheduledNotificationsAsync();
    }
    ```
