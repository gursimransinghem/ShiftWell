    ```typescript
    export interface BriefContext {
      userId: string;
      weekStartISO: string;           // Monday date (yyyy-MM-dd)
      adherencePercent: number;       // 0-100, past 7 nights
      sleepDebtHours: number;         // Current debt from debt engine
      avgDiscrepancyMinutes: number;  // Mean |delta.startMinutes| past 7 nights
      upcomingShifts: {               // Next 7 days
        dateISO: string;
        type: 'day' | 'evening' | 'night' | 'off';
      }[];
      hardTransitionDays: number;     // Count of transitions >= 8h circadian phase shift
    }

    export interface WeeklyBrief {
      id: string;                     // UUID
      weekStartISO: string;
      generatedAtISO: string;
      text: string;                   // Claude's output (max 150 words)
      recommendation: string;         // The single actionable recommendation
      adherencePercent: number;       // Snapshot at generation time
      model: string;                  // e.g. "claude-haiku-3-5"
      tokensUsed: number;
      passedGuardrails: boolean;
    }

    export interface BriefGenerationResult {
      success: boolean;
      brief?: WeeklyBrief;
      error?: string;
      guardrailFailure?: string;
    }

    export class ClaudeAPIError extends Error {
      constructor(public statusCode: number, message: string) {
        super(message);
        this.name = 'ClaudeAPIError';
      }
    }
    ```

    **Step 2: Create `src/lib/ai/claude-client.ts`**

    - Read EXPO_PUBLIC_ANTHROPIC_API_KEY from environment
    - Export `generateCompletion(systemPrompt: string, userMessage: string, model?: string): Promise<{ text: string; tokensUsed: number }>`
    - Use `fetch` to POST to `https://api.anthropic.com/v1/messages` (Anthropic Messages API)
    - Headers: `x-api-key`, `anthropic-version: 2023-06-01`, `content-type: application/json`
    - Body: `{ model: model ?? 'claude-haiku-4-5', max_tokens: 300, system: systemPrompt, messages: [{ role: 'user', content: userMessage }] }`
    - On 429: throw ClaudeAPIError(429, 'Rate limit exceeded — retry after 60s')
    - On 4xx/5xx: throw ClaudeAPIError(statusCode, response body message)
    - Return `{ text: response.content[0].text, tokensUsed: response.usage.input_tokens + response.usage.output_tokens }`

    **`buildUserMessage(ctx: BriefContext): string`**
    Construct structured user message from BriefContext:
    - "Past 7 nights: {adherencePercent}% adherence, {sleepDebtHours}h sleep debt"
    - "Average bedtime discrepancy: {avgDiscrepancyMinutes} minutes"
    - "Upcoming week: {shift types summary} with {hardTransitionDays} hard transitions"
    - "Generate a personalized Monday morning sleep summary."

    **`validateGuardrails(text: string): { pass: boolean; violation?: string }`**
    Check text against prohibited patterns derived from SAFETY-GUARDRAILS.md Section 1:
    - Diagnosis terms: /\b(disorder|syndrome|disease|condition|diagnos)/i → violation: "diagnosis"
    - Medication terms: /\b(melatonin|medication|supplement|drug|prescri|dosage|mg\b)/i → violation: "medication"
    - Prognosis terms: /\b(will cause|leads to|results in|increases risk of)/i → violation: "prognosis"
    - Emergency terms: /\b(emergency|urgent|immediately see|call 911|ER|hospital)/i → violation: "emergency"

    **`extractRecommendation(text: string): string`**
    Return last sentence that starts with an imperative verb (Try, Consider, Aim, Move, Add, Prioritize, Avoid). Fallback: return last sentence of text.

    **`generateWeeklyBrief(ctx: BriefContext): Promise<BriefGenerationResult>`**
    Full pipeline:
    1. Build system prompt and user message
    2. Call `generateCompletion(systemPrompt, userMessage)`
    3. Run `validateGuardrails(text)` — if fail, return `{ success: false, guardrailFailure: violation }`
    4. Extract recommendation
    5. Construct `WeeklyBrief` with UUID (`crypto.randomUUID()` or uuid package), timestamps, token count
    6. Return `{ success: true, brief }`
    7. On ClaudeAPIError: return `{ success: false, error: error.message }`
