    /**
     * Referral deep link service.
     * Generates shareable links that credit the referring user.
     * Uses expo-linking for deep link handling on install.
     *
     * Format: https://shiftwell.app/join?ref={BASE36_CODE}
     * Code: first 8 chars of base36-encoded userId hash (deterministic)
     */
    import AsyncStorage from '@react-native-async-storage/async-storage';

    const REFERRAL_BASE_URL = 'https://shiftwell.app/join';
    const REFERRAL_STORAGE_KEY = 'shiftwell-referral-code';

    /** Generate a deterministic referral code from userId (not a UUID — must be reproducible) */
    function generateReferralCode(userId: string): string {
      // Simple hash: sum char codes, encode in base36, take first 8 chars uppercase
      let hash = 0;
      for (let i = 0; i < userId.length; i++) {
        hash = ((hash << 5) - hash) + userId.charCodeAt(i);
        hash |= 0; // Convert to 32bit integer
      }
      return Math.abs(hash).toString(36).toUpperCase().padStart(8, '0').slice(0, 8);
    }

    export function generateReferralLink(userId: string): string {
      const code = generateReferralCode(userId);
      return `${REFERRAL_BASE_URL}?ref=${code}`;
    }

    export function decodeReferralCode(code: string): string {
      // Code is deterministic from userId — not reversible in v1 (lookup would need Supabase)
      // In v1: store the code for attribution; Supabase resolves code -> userId on backend
      return code;
    }

    /** Store referral code when user installs via referral link */
    export async function trackReferralInstall(code: string): Promise<void> {
      await AsyncStorage.setItem(REFERRAL_STORAGE_KEY, code);
    }

    /** Get stored referral code (if user installed via referral) */
    export async function getStoredReferralCode(): Promise<string | null> {
      return AsyncStorage.getItem(REFERRAL_STORAGE_KEY);
    }
    ```

    /**
     * A/B Testing Framework — deterministic user bucketing.
     * Uses a hash of (experimentId + userId) to assign stable variants.
     * No external SDK required — pure deterministic bucketing.
     */
    import AsyncStorage from '@react-native-async-storage/async-storage';

    type Variant = 'A' | 'B' | 'control' | 'experiment';

    const EXPERIMENTS: Record<string, { variants: Variant[]; description: string }> = {
      onboarding: {
        variants: ['A', 'B'],
        description: 'Onboarding flow variant — A: standard, B: science-first',
      },
      paywall: {
        variants: ['control', 'experiment'],
        description: 'Paywall pricing — control: $29.99/yr, experiment: $24.99/yr',
      },
    };

    function hashString(str: string): number {
      let hash = 5381;
      for (let i = 0; i < str.length; i++) {
        hash = ((hash << 5) + hash) + str.charCodeAt(i);
        hash |= 0;
      }
      return Math.abs(hash);
    }

    /**
     * Get deterministic variant for a user in an experiment.
     * Same experimentId + userId always returns same variant.
     */
    export function getVariant(experimentId: string, userId: string): Variant {
      const experiment = EXPERIMENTS[experimentId];
      if (!experiment) {
        console.warn(`[ab-testing] Unknown experiment: ${experimentId}`);
        return 'A';
      }
      const hash = hashString(`${experimentId}:${userId}`);
      const index = hash % experiment.variants.length;
      return experiment.variants[index];
    }

    /** Record a conversion event for the active experiment */
    export async function trackConversion(
      experimentId: string,
      userId: string,
      eventType: 'onboarding_complete' | 'trial_start' | 'purchase' = 'onboarding_complete'
    ): Promise<void> {
      const variant = getVariant(experimentId, userId);
      const event = {
        experimentId,
        userId,
        variant,
        eventType,
        timestamp: new Date().toISOString(),
      };
      // Persist conversion event locally; batch-send to Supabase in Phase 19
      const key = `ab-conversion-${experimentId}-${userId}`;
      await AsyncStorage.setItem(key, JSON.stringify(event));
    }
    ```

    ```typescript
    /**
     * Re-engagement notification sequence.
     * Fires at D+1, D+3, D+7 after last app open for inactive users.
     * Cancels automatically when user opens the app.
     *
     * GRO-03: D1, D3, D7 re-engagement sequence.
     */
    import * as Notifications from 'expo-notifications';

    const RE_ENGAGEMENT_IDS = ['re-engage-d1', 're-engage-d3', 're-engage-d7'];

    const MESSAGES = [
      { id: 're-engage-d1', delay: 24 * 60 * 60, title: 'Your shift is tomorrow', body: 'Quick check — is your sleep plan ready for your next shift?' },
      { id: 're-engage-d3', delay: 3 * 24 * 60 * 60, title: 'Your sleep debt is growing', body: 'You haven\'t checked in for 3 days. Your circadian plan is waiting.' },
      { id: 're-engage-d7', delay: 7 * 24 * 60 * 60, title: 'Don\'t lose your streak', body: '7 days of adaptive data helps ShiftWell learn your patterns. Come back!' },
    ];

    /** Schedule D1, D3, D7 re-engagement notifications from now */
    export async function scheduleReengagementSequence(): Promise<void> {
      await cancelReengagementSequence();
      for (const msg of MESSAGES) {
        await Notifications.scheduleNotificationAsync({
          identifier: msg.id,
          content: { title: msg.title, body: msg.body, data: { type: 're-engagement' } },
          trigger: { type: Notifications.SchedulableTriggerInputTypes.TIME_INTERVAL, seconds: msg.delay },
        });
      }
    }

    /** Cancel all re-engagement notifications (call on app foreground) */
    export async function cancelReengagementSequence(): Promise<void> {
      for (const id of RE_ENGAGEMENT_IDS) {
        await Notifications.cancelScheduledNotificationAsync(id).catch(() => {});
      }
    }
    ```

    export interface PaywallVariant {
      annualPrice: string;      // e.g. '$24.99'
      annualPriceNumeric: number; // e.g. 24.99
      monthlyEquivalent: string; // e.g. '$2.08/mo'
      variantId: 'control' | 'experiment';
    }

    const VARIANTS: Record<string, PaywallVariant> = {
      control: { annualPrice: '$29.99', annualPriceNumeric: 29.99, monthlyEquivalent: '$2.50/mo', variantId: 'control' },
      experiment: { annualPrice: '$24.99', annualPriceNumeric: 24.99, monthlyEquivalent: '$2.08/mo', variantId: 'experiment' },
    };

    export function getPaywallVariant(userId: string): PaywallVariant {
      const variant = getVariant('paywall', userId) as 'control' | 'experiment';
      return VARIANTS[variant] ?? VARIANTS.control;
    }
