    ```typescript
    // Title builder for ALL block types (writeSleepBlock only handles main-sleep/nap)
    export function buildPlanBlockTitle(block: PlanBlock): string {
      const timeStr = format(block.start, 'h:mm a').replace('am', 'AM').replace('pm', 'PM');
      switch (block.type) {
        case 'main-sleep': return `Sleep — ${timeStr}`;
        case 'nap':        return `Nap — ${timeStr}`;
        case 'wind-down':  return `Wind Down — ${timeStr}`;
        case 'wake':       return `Wake — ${timeStr}`;
        case 'caffeine-cutoff': return `Caffeine Cutoff — ${timeStr}`;
        case 'meal-window':    return `Meal Window — ${timeStr}`;
        case 'light-seek':     return `Light: Seek — ${timeStr}`;
        case 'light-avoid':    return `Light: Avoid — ${timeStr}`;
        default:               return `${block.label} — ${timeStr}`;
      }
    }

    // Write a single plan block to an Apple calendar (any block type)
    export async function writePlanBlock(block: PlanBlock, calendarId: string): Promise<string>
    // Use ExpoCalendar.createEventAsync directly (same as writeSleepBlock in calendar-service.ts)
    // Title = buildPlanBlockTitle(block)

    // Simple change detection
    export function blockChanged(a: PlanBlock, b: PlanBlock): boolean {
      return a.start.getTime() !== b.start.getTime() || a.end.getTime() !== b.end.getTime();
    }

    // Diff + sync plan blocks to ShiftWell Apple calendar
    export async function writeChangedBlocks(
      oldPlan: SleepPlan | null,
      newPlan: SleepPlan,
      calStore: { shiftWellCalendarId: string | null; eventIdMap: Record<string, string>; mapEventId: (calId: string, planId: string) => void; removeEventId: (calId: string) => void },
    ): Promise<void>
    ```

    writeChangedBlocks algorithm:
    1. If `calStore.shiftWellCalendarId` is null → return early (no ShiftWell calendar yet)
    2. Build inverse map: `planToEventId = Object.fromEntries(Object.entries(calStore.eventIdMap).map(([calId, planId]) => [planId, calId]))`
    3. Build set of new block IDs: `newBlockIds = new Set(newPlan.blocks.map(b => b.id))`
    4. Delete removed blocks: for each `[calEventId, planBlockId]` in `calStore.eventIdMap` where `!newBlockIds.has(planBlockId)` → `deleteSleepBlock(calEventId, shiftWellCalId)` then `calStore.removeEventId(calEventId)`
    5. For each block in `newPlan.blocks`:
       - Get `existingEventId = planToEventId[block.id]`
       - If exists: find oldBlock in oldPlan; if `blockChanged(oldBlock, block)` → `updateSleepBlock(existingEventId, block, shiftWellCalId)` (updateSleepBlock already uses buildSleepBlockTitle internally — for non-sleep types, call `ExpoCalendar.updateEventAsync` directly with `buildPlanBlockTitle(block)`)
       - If not exists: `newEventId = await writePlanBlock(block, shiftWellCalId)` → `calStore.mapEventId(newEventId, block.id)`
    6. Best-effort: wrap each calendar call in try/catch, log warn on failure, continue

    ```typescript
    let regenerateTimer: ReturnType<typeof setTimeout> | null = null;
    function debouncedRegenerate(opts?: { isCircadianReset?: boolean }, ms = 500): void {
      if (regenerateTimer) clearTimeout(regenerateTimer);
      regenerateTimer = setTimeout(() => {
        usePlanStore.getState().regeneratePlan(opts);
        regenerateTimer = null;
      }, ms);
    }
    ```
