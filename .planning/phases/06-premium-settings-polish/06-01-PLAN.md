**Step 2 — premium-service.ts:** Add `startTrial()` and `getTrialStartDate()` utilities:

```typescript
const TRIAL_STORAGE_KEY = 'shiftwell-trial-start';

export async function startTrial(): Promise<void> {
  const existing = await AsyncStorage.getItem(TRIAL_STORAGE_KEY);
  if (!existing) {
    await AsyncStorage.setItem(TRIAL_STORAGE_KEY, new Date().toISOString());
  }
}

export async function getTrialStartDate(): Promise<Date | null> {
  const stored = await AsyncStorage.getItem(TRIAL_STORAGE_KEY);
  return stored ? new Date(stored) : null;
}
```

Import `AsyncStorage from '@react-native-async-storage/async-storage'`.
