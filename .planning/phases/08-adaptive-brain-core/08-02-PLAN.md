    ```typescript
    async function run() {
      setIsLoading(true);
      try {
        const today = new Date();
        const todayISO = format(today, 'yyyy-MM-dd');

        // Daily debounce: skip if already ran today (BRAIN-01)
        const lastRun = await AsyncStorage.getItem(ADAPTIVE_LAST_RUN_KEY);
        if (lastRun === todayISO) {
          return;  // already ran today — skip
        }

        // ... existing HealthKit + context logic unchanged ...

        // Mark as run AFTER successful completion (so failures retry)
        await AsyncStorage.setItem(ADAPTIVE_LAST_RUN_KEY, todayISO);
      } catch (err) {
        console.warn('[useAdaptivePlan] Failed to assemble context:', err);
      } finally {
        if (!cancelled) setIsLoading(false);
      }
    }
    ```
