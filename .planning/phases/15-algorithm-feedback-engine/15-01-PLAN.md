    **Step 2: Create feedback-engine.ts**

    ```typescript
    /**
     * Algorithm Feedback Engine — HealthKit Closed Loop
     *
     * Adjusts planned sleep windows based on systematic deviation between
     * planned and actual sleep timing (from HealthKit data).
     *
     * Algorithm: Exponential Moving Average (EMA) of bedtime deviations.
     * Scientific basis: Process S homeostatic calibration (Borbely 1982),
     * EMA smoothing for noisy actigraphy (Phillips et al. 2017).
     *
     * Phase 13 research defines all constants used here:
     * - SMOOTHING_ALPHA = 0.3 (conservative adaptation)
     * - DEAD_ZONE_MINUTES = 20 (Apple Watch ±20 min accuracy floor)
     * - MAX_DELTA_MINUTES = 30 (max per-cycle adjustment)
     * - MIN_NIGHTS_REQUIRED = 3 (minimum data before feedback activates)
     * - MAX_GAP_NIGHTS = 3 (consecutive missing nights triggers pause)
     * - CONVERGENCE_TARGET = 15 (avg deviation below this = converged)
     */

    import type { SleepComparison } from '../healthkit/sleep-comparison';
    import type { CircadianProtocol, FeedbackResult } from './types';

    const SMOOTHING_ALPHA = 0.3;
    const DEAD_ZONE_MINUTES = 20;
    const MAX_DELTA_MINUTES = 30;
    const MIN_NIGHTS_REQUIRED = 3;
    const MAX_GAP_NIGHTS = 3;
    const CONVERGENCE_TARGET = 15;
    const STALL_THRESHOLD = 30; // if avg deviation > 30 after 14 nights = stalled

    function clamp(value: number, min: number, max: number): number {
      return Math.max(min, Math.min(max, value));
    }

    function computeEMA(deviations: number[], alpha: number): number {
      if (deviations.length === 0) return 0;
      return deviations.reduce((ema, d) => alpha * d + (1 - alpha) * ema, deviations[0]);
    }

    function countConsecutiveMissingNights(history: SleepComparison[]): number {
      let count = 0;
      for (let i = history.length - 1; i >= 0; i--) {
        if (history[i].actual === null) count++;
        else break;
      }
      return count;
    }

    /**
     * Compute the feedback offset to apply to the sleep plan.
     *
     * @param discrepancyHistory - Last 30 nights of plan-vs-actual comparisons (from Phase 14)
     * @param protocol - Active circadian protocol (null if none)
     * @param previousOffset - The currently applied offset (to accumulate EMA delta onto)
     */
    export function computeFeedbackOffset(
      discrepancyHistory: SleepComparison[],
      protocol: CircadianProtocol | null,
      previousOffset: { bedtimeMinutes: number; wakeMinutes: number } = { bedtimeMinutes: 0, wakeMinutes: 0 }
    ): FeedbackResult {
      const inactive = (reason: string): FeedbackResult => ({
        adjustedBedtimeOffsetMinutes: previousOffset.bedtimeMinutes,
        adjustedWakeOffsetMinutes: previousOffset.wakeMinutes,
        feedbackActive: false,
        feedbackReason: reason,
        smoothedBedtimeDeviation: 0,
        convergenceStatus: 'insufficient_data',
      });

      // Guard 1: Insufficient data
      const validNights = discrepancyHistory.filter(n => n.actual !== null);
      if (validNights.length < MIN_NIGHTS_REQUIRED) {
        return inactive(`Need ${MIN_NIGHTS_REQUIRED} nights of HealthKit data (have ${validNights.length})`);
      }

      // Guard 2: Active circadian protocol takes priority
      if (protocol && protocol.transitionType !== 'none') {
        return {
          ...inactive(`Feedback paused — protocol active (${protocol.transitionType})`),
          convergenceStatus: 'converging', // will resume after protocol ends
        };
      }

      // Guard 3: Data gap (3+ consecutive missing nights)
      const gapNights = countConsecutiveMissingNights(discrepancyHistory);
      if (gapNights >= MAX_GAP_NIGHTS) {
        return inactive(`Feedback paused — ${gapNights} consecutive nights without HealthKit data`);
      }

      // Compute EMA on bedtime deviations from valid nights (most recent 7 for responsiveness)
      const recentNights = validNights.slice(-7);
      const bedtimeDeviations = recentNights.map(n => n.bedtimeDeviationMinutes);
      const wakeDeviations = recentNights.map(n => n.wakeDeviationMinutes);

      const smoothedBedtime = computeEMA(bedtimeDeviations, SMOOTHING_ALPHA);
      const smoothedWake = computeEMA(wakeDeviations, SMOOTHING_ALPHA);

      // Apply dead zone
      const bedtimeDelta = Math.abs(smoothedBedtime) < DEAD_ZONE_MINUTES
        ? 0
        : clamp(smoothedBedtime * 0.5, -MAX_DELTA_MINUTES, MAX_DELTA_MINUTES);

      const wakeDelta = Math.abs(smoothedWake) < DEAD_ZONE_MINUTES
        ? 0
        : clamp(smoothedWake * 0.5, -MAX_DELTA_MINUTES, MAX_DELTA_MINUTES);

      const newBedtimeOffset = previousOffset.bedtimeMinutes + bedtimeDelta;
      const newWakeOffset = previousOffset.wakeMinutes + wakeDelta;

      // Determine convergence status
      const avgAbsDeviation = bedtimeDeviations.reduce((s, d) => s + Math.abs(d), 0) / bedtimeDeviations.length;
      let convergenceStatus: FeedbackResult['convergenceStatus'];
      if (avgAbsDeviation <= CONVERGENCE_TARGET) {
        convergenceStatus = 'converged';
      } else if (validNights.length >= 14 && avgAbsDeviation > STALL_THRESHOLD) {
        convergenceStatus = 'stalled';
      } else {
        convergenceStatus = 'converging';
      }

      return {
        adjustedBedtimeOffsetMinutes: newBedtimeOffset,
        adjustedWakeOffsetMinutes: newWakeOffset,
        feedbackActive: true,
        feedbackReason: convergenceStatus === 'converged'
          ? `Converged — avg deviation ${avgAbsDeviation.toFixed(0)} min`
          : `Adjusting — avg deviation ${avgAbsDeviation.toFixed(0)} min (target: ${CONVERGENCE_TARGET} min)`,
        smoothedBedtimeDeviation: smoothedBedtime,
        convergenceStatus,
      };
    }
    ```
