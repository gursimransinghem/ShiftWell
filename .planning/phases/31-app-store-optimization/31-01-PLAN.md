**src/i18n/index.ts — i18n provider:**
```typescript
import { en } from './en';
import { es } from './es';
import { getLocales } from 'expo-localization';

type SupportedLocale = 'en' | 'es';

const translations = { en, es };

export function getLocale(): SupportedLocale {
  const deviceLocale = getLocales()[0]?.languageCode ?? 'en';
  return deviceLocale.startsWith('es') ? 'es' : 'en';
}

export function t(key: string): string
// Supports dot notation: t('today.recoveryScore') → "Recovery Score" (en) or "Puntuación de recuperación" (es)
// Falls back to English if key missing in Spanish
// Uses getLocale() on every call (no caching needed — locale doesn't change during session)

export function useTranslation() {
  const locale = getLocale();
  return { t: (key: string) => t(key), locale };
}
```
